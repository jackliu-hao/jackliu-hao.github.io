1、编写配置文件

cryptogen :生成对应的证书的工具

cryptogen showtemplate > a.yaml 输出配置文件的模板到a.yaml

# ---------------------------------------------------------------------------
# "OrdererOrgs" - Definition of organizations managing orderer nodes
# ---------------------------------------------------------------------------
#排序节点对应的组织 ，可以集群
OrdererOrgs:
  # ---------------------------------------------------------------------------
  # Orderer
  # ---------------------------------------------------------------------------
  - Name: Orderer  # 排序节点组织的名字 ，这里配置的是组织
    Domain: example.com  #根域名，排序节点组织的根域名
    EnableNodeOUs: true # 

    # ---------------------------------------------------------------------------
    # "Specs" - See PeerOrgs below for complete description
    # ---------------------------------------------------------------------------
    #使用Specs，二级域名可以自己指定
    Specs:
      - Hostname: orderer   # 访问这台order对饮的域名是 ： order.example.com 
      #可能有其他的Hostname: orderer1 ，在访问的时候的域名是:  order1.example.com

# ---------------------------------------------------------------------------
# "PeerOrgs" - Definition of organizations managing peer nodes
# ---------------------------------------------------------------------------
PeerOrgs:
  #此配置文件共有两个组织
  # ---------------------------------------------------------------------------
  # org_go
  # ---------------------------------------------------------------------------
  - Name: org_go    #第一个组织的名字 ， 自己的指定 
    Domain: org_go.example.com   #访问第一个组织用的根域名
    EnableNodeOUs: true    #是否支生成证书
   
    # ---------------------------------------------------------------------------
    Template:      #模板 ，根据默认的规则生成2个peer存储数据的节点
      Count: 2  #1. peer0.org_go.example.com     peer1.org_go.example.com 
    Users:   #创建的普通用户的个数
      Count: 3
  # ---------------------------------------------------------------------------
  # org_cpp: See "org_go" for full specification
  # ---------------------------------------------------------------------------
  - Name: org_cpp
    Domain: org_cpp.example.com
    EnableNodeOUs: true
    Template:
      Count: 2
    Users:
      Count: 1
